CSS

*{
margin: 0;
padding: 0;
box-sizing: border-box;
width: 100%;
}

global reset

display: flex;

convert to flex box. Options: flex, inline-flex

flex-grow: 3;

how much space to absorb when growing

flex-shrink: 3;

how much space to give up when shrinking

flex-direction: column;

direction of flex. options: column or row

flex-wrap: wrap;

allow overflow of flex items. Options: wrap, nowrap

flex-basis: 200px;

specify initial size of flex

flex: 2 1 200px;

specify flex-grow, flex-shrink and flex-basis in one line

flex-flow: wrap col;

specify flex-wrap and flex-direction in one line

justify-content: space-around;

put space between flex items along main axis. Options: space-between, center, flex-start, flex-end

align-items: center;

put space between items along cross axis. Options: space-between, center, flex-start, flex-end

display: grid;

converts div to grid

grid-template: 200px 200px / 25% 25% 2fr 1fr;

determines number, width and height of rows and columns of grid

gap: 10px 15px;

determine space between the row and column

grid-area: 1/1/span 2/span 2;

define size of item in grid - row start, column start, row end, column end

GIT

git init

turn directory into empty git repo

git add .

add all files into staging area for commiting

git add fileName.ext

add fileName.ext into staging area for commiting

git commit -m "my commit message"

commit staged files to repo

git push --all

push committed files to repo

git branch branchName

create new branch

git checkout branchName

switch branches to start working in new branch

git merge branchName

merge branches together

git remote add origin remoteName remoteURL

connect to a remote repo

git clone remoteURL

git clone

git pull branchName repoName

get the latest verion of repo
